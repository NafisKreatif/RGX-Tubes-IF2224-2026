program Test;

const
  Min == 1;
  Max == 5;
  StartChar == 'a';
  Label == 'level';

type
  SmallRange == 1 .. 5;
  Direction == (Left, Right, Center);
  Pair == record
    first, second: integer;
    tag: char
  end;
  NumberList == array[1 .. 5] of integer;

var
  i, total, middle: integer;
  numbers: NumberList;
  pairValue: Pair;
  isValid, isDone: boolean;

procedure PrintValue(value: integer; data: array[1 .. 5] of integer);
begin
  writeln('value = ', value)
end;

function MirrorIndex(index: integer): integer;
var
  result: integer;
begin
  result := Max - index + Min;
  MirrorIndex := result
end;

begin
  i := Min;
  total := 0;
  middle := (Min + Max) div 2;
  numbers[1, 2, 3] := 9;
  pairValue.first := numbers[1] * 2;
  pairValue.second := pairValue.first / 2;
  pairValue.tag := StartChar;
  isValid := true;
  isDone := false;
  if not (pairValue.first == pairValue.second) then isValid := false else PrintValue(pairValue.first, numbers);
  while i <= Max do
    i := i + 1;
  repeat
    total := total + MirrorIndex(i) mod 3;
    i := i - 1
  until i < Min;
  for i := Min to Max do PrintValue(i, numbers);
  for i := Max downto Min do total := total + i;
  case total of
    0: total := total + 1;
    1, 2: total := total - 1;
    3: PrintValue(total, numbers)
  end;
  isDone := isValid and true or false
end.
